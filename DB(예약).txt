DROP TABLE RESERVATION
CREATE TABLE RESERVATION (
	ROOM_NO NUMBER,
	A_NUM NUMBER NOT NULL,
	C_NUM NUMBER,
	I_NUM NUMBER,
	CHECKIN_DATE DATE NOT NULL,
	CHECKOUT_DATE DATE NOT NULL,
	RE_NUM NUMBER NOT NULL,
      CONSTRAINT PK_RNO PRIMARY KEY (ROOM_NO)
);

COMMENT ON COLUMN RESERV.ROOM_NO IS '숙소 번호';
COMMENT ON COLUMN RESERV.A_NUM IS '성인';
COMMENT ON COLUMN RESERV.C_NUM IS '어린이';
COMMENT ON COLUMN RESERV.I_NUM IS '유아';
COMMENT ON COLUMN RESERV.CHECKIN_DATE IS '체크인 날짜';
COMMENT ON COLUMN RESERV.CHECKOUT_DATE IS '체크아웃 날짜';
COMMENT ON COLUMN RESERV.RE_NUM IS '예약 번호';

ALTER TABLE RESERVATION
ADD CONSTRAINT FK_RE_UID FOREIGN KEY (USER_ID) REFERENCES MEMBER ON DELETE CASCADE;
